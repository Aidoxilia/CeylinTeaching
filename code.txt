export default function App() {
  const [index, setIndex] = useState(0);
  const [selected, setSelected] = useState(null);
  const [isAnswered, setIsAnswered] = useState(false);
  const [score, setScore] = useState(0);
  const [finished, setFinished] = useState(false);

  const current = examData[index];
  const progress = (index / examData.length) * 100;

  const TopicIcon = ({ type }) => {
    switch (type) {
      case 'fluids': return <Droplets size={18} className="text-blue-500" />;
      case 'thermal': return <Thermometer size={18} className="text-red-500" />;
      case 'materials': return <Activity size={18} className="text-green-500" />;
      default: return <BookOpen size={18} />;
    }
  };

  const handleChoice = (idx) => {
    if (isAnswered) return;
    setSelected(idx);
    setIsAnswered(true);
    if (idx === current.correctAnswer) setScore(prev => prev + 1);
  };

  const next = () => {
    if (index + 1 < examData.length) {
      setIndex(index + 1);
      setSelected(null);
      setIsAnswered(false);
    } else {
      setFinished(true);
    }
  };

  if (finished) {
    return (
      <div className="min-h-screen bg-slate-50 flex items-center justify-center p-6">
        <div className="max-w-md w-full bg-white rounded-3xl shadow-xl p-8 text-center">
          <div className="w-20 h-20 bg-blue-100 text-blue-600 rounded-full flex items-center justify-center mx-auto mb-6">
            <CheckCircle2 size={40} />
          </div>
          <h2 className="text-2xl font-bold mb-2 text-slate-800">Practice Complete</h2>
          <div className="bg-slate-50 py-6 rounded-2xl my-6">
            <p className="text-4xl font-black text-blue-600">{score} / {examData.length}</p>
            <p className="text-slate-500 mt-2 font-medium">Final Score</p>
          </div>
          <button onClick={() => window.location.reload()} className="w-full bg-blue-600 text-white py-4 rounded-xl font-bold flex items-center justify-center gap-2">
            <RotateCcw size={20} /> Restart Session
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-slate-50 flex flex-col items-center py-10 px-4">
      <div className="max-w-2xl w-full">
        <header className="mb-8 flex justify-between items-end">
          <div>
            <h1 className="text-2xl font-bold text-slate-900">BTEC Unit 5 Physics</h1>
            <p className="text-slate-500 text-sm font-medium">31627H/1P Exam Prep</p>
          </div>
          <div className="bg-blue-600 text-white px-4 py-1 rounded-full text-sm font-bold">
            Score: {score}
          </div>
        </header>

        <div className="w-full bg-slate-200 h-2 rounded-full mb-8 overflow-hidden">
          <div className="bg-blue-600 h-full transition-all duration-300" style={{ width: `${progress}%` }} />
        </div>

        <main className="bg-white rounded-3xl shadow-md border border-slate-100 overflow-hidden">
          <div className="p-6 md:p-8">
            <div className="flex gap-2 mb-4">
              <span className="flex items-center gap-1 bg-slate-100 px-3 py-1 rounded-lg text-xs font-bold text-slate-600 uppercase">
                <TopicIcon type={current.type} /> {current.topic}
              </span>
              <span className="bg-amber-100 text-amber-700 px-3 py-1 rounded-lg text-xs font-bold">
                {current.paper}
              </span>
            </div>
            <h2 className="text-xl font-semibold text-slate-800 leading-snug">
              {current.question}
            </h2>
          </div>

          <div className="px-6 md:px-8 pb-8 space-y-3">
            {current.options.map((opt, i) => (
              <button
                key={i}
                disabled={isAnswered}
                onClick={() => handleChoice(i)}
                className={`w-full text-left p-4 rounded-xl border-2 transition-all flex justify-between items-center ${
                  isAnswered 
                    ? i === current.correctAnswer 
                      ? "bg-green-50 border-green-500 text-green-800" 
                      : i === selected 
                        ? "bg-red-50 border-red-500 text-red-800"
                        : "bg-white border-slate-100 opacity-50"
                    : "bg-white border-slate-200 hover:border-blue-400"
                }`}
              >
                <span className="font-medium">{opt}</span>
                {isAnswered && i === current.correctAnswer && <CheckCircle2 size={18} />}
                {isAnswered && i === selected && i !== current.correctAnswer && <XCircle size={18} />}
              </button>
            ))}

            {isAnswered && (
              <div className="mt-6 p-5 bg-blue-50 rounded-2xl border border-blue-100 animate-in fade-in slide-in-from-bottom-2 duration-300">
                <p className="text-sm text-blue-900 leading-relaxed italic">
                  <strong>Explanation:</strong> {current.explanation}
                </p>
                <button 
                  onClick={next}
                  className="mt-4 w-full bg-slate-900 text-white py-3 rounded-xl font-bold flex items-center justify-center gap-2 hover:bg-slate-800 transition-colors"
                >
                  {index + 1 === examData.length ? "Finish Quiz" : "Next Question"}
                  <ArrowRight size={18} />
                </button>
              </div>
            )}
          </div>
        </main>
      </div>
    </div>
  );
}
